132  string preprocess(const string& input) {
133  string output = Strings::rtrim(input, ';');
134  if (output.find("))") != string::npos) {
135  size_t stpos = output.rfind(':');
136  string sub = output.substr(stpos + 1);
137  sub = Strings::trim(Strings::rtrim(sub, '}'));
138  vector<string> conds = Strings::split(sub, "&&");
139 
140  // 1) 1st pass: identify assignments...
141  map<string, string> assigns;
142  map<string, size_t> locs;
143  string lhs, rhs;
144  size_t loc;
145 
146  for (const string& cond : conds) {
147  // Check for UF and equality statement
148  if (cond.find('(') != string::npos && cond.find(" = ") != string::npos) {
149  loc = output.find(cond);
150  vector<string> elems = Strings::split(cond, " = ");
151  lhs = elems[0];
152  rhs = elems[1];
153 
154  // Which one is the function, right or left?
155  if (lhs.find('(') != string::npos) {
156  string swap = lhs;
157  lhs = rhs;
158  rhs = swap;
159  }
160  assigns[lhs] = rhs;
161  locs[lhs] = loc;
162  }
163  }
164 
165  // 2) 2nd pass: replace assignments within conditions
166  for (auto itr = assigns.begin(); itr != assigns.end(); ++itr) {
167  lhs = itr->first;
168  rhs = itr->second;
169  loc = locs[lhs];
170 
171  size_t pos = output.find(rhs, stpos + 1);
172  if (pos <= loc) {
173  pos = output.find(rhs, loc + 1);
174  }
175 
176  while((pos = output.find(rhs, pos)) != string::npos) {
177  output.replace(pos, rhs.length(), lhs);
178  pos += lhs.length(); // In case 'to' contains 'from', like replacing 'x' with 'yx'
179  }
180  }
181  }
182 
183  return output;
184  }
