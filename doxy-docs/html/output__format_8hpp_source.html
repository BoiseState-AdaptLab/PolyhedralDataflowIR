224 template<typename CharT>
225 void escape_string(const CharT* s,
226  size_t length,
227  const basic_output_format<CharT>& format,
228  buffered_ostream<CharT>& os)
229 {
230  const CharT* begin = s;
231  const CharT* end = s + length;
232  for (const CharT* it = begin; it != end; ++it)
233  {
234  CharT c = *it;
235  switch (c)
236  {
237  case '\\':
238  os.put('\\');
239  os.put('\\');
240  break;
241  case '"':
242  os.put('\\');
243  os.put('\"');
244  break;
245  case '\b':
246  os.put('\\');
247  os.put('b');
248  break;
249  case '\f':
250  os.put('\\');
251  os.put('f');
252  break;
253  case '\n':
254  os.put('\\');
255  os.put('n');
256  break;
257  case '\r':
258  os.put('\\');
259  os.put('r');
260  break;
261  case '\t':
262  os.put('\\');
263  os.put('t');
264  break;
265  default:
266  uint32_t u(c >= 0 ? c : 256 + c);
267  if (format.escape_solidus() && c == '/')
268  {
269  os.put('\\');
270  os.put('/');
271  }
272  else if (is_control_character(u) || format.escape_all_non_ascii())
273  {
274  // convert utf8 to codepoint
275  uint32_t cp = json_char_traits<CharT, sizeof(CharT)>::convert_char_to_codepoint(it, end);
276  if (is_non_ascii_character(cp) || is_control_character(u))
277  {
278  if (cp > 0xFFFF)
279  {
280  cp -= 0x10000;
281  uint32_t first = (cp >> 10) + 0xD800;
282  uint32_t second = ((cp & 0x03FF) + 0xDC00);
283 
284  os.put('\\');
285  os.put('u');
286  os.put(to_hex_character(first >> 12 & 0x000F));
287  os.put(to_hex_character(first >> 8 & 0x000F));
288  os.put(to_hex_character(first >> 4 & 0x000F));
289  os.put(to_hex_character(first & 0x000F));
290  os.put('\\');
291  os.put('u');
292  os.put(to_hex_character(second >> 12 & 0x000F));
293  os.put(to_hex_character(second >> 8 & 0x000F));
294  os.put(to_hex_character(second >> 4 & 0x000F));
295  os.put(to_hex_character(second & 0x000F));
296  }
297  else
298  {
299  os.put('\\');
300  os.put('u');
301  os.put(to_hex_character(cp >> 12 & 0x000F));
302  os.put(to_hex_character(cp >> 8 & 0x000F));
303  os.put(to_hex_character(cp >> 4 & 0x000F));
304  os.put(to_hex_character(cp & 0x000F));
305  }
306  }
307  else
308  {
309  os.put(c);
310  }
311  }
312  else if (format.escape_solidus() && c == '/')
313  {
314  os.put('\\');
315  os.put('/');
316  }
317  else
318  {
319  os.put(c);
320  }
321  break;
322  }
323  }
324 }
