169  // must be postive (i.e. b > 0 below), otherwise function undefined
170  // Note:
171  // ceil(a, b) = -floor(-a, b) or floor(a+b-1, b) or floor(a-1, b)+1
172  // mod(a, b) = a-b*floor(a, b)
173  // where result must lie in range [0,b)
174  // floor(a, b) = a/b if a >= 0
175  // (a-b+1)/b if a < 0
176  // where native '/' operator behaves as 5/2 = 2, (-5)/2 = -2
177  //---------------------------------------------------------------------------

179  virtual CG_outputRepr *CreateIntegerMod(CG_outputRepr *lop, CG_outputRepr *rop) const {
180  CG_outputRepr *lop2 = CreateCopy(lop);
