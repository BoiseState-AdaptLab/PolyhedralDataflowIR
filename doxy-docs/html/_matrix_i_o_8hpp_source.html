541

546  } while (num_items_read != 3);
547 
548  return 0;

699  int comp = 0;

701  coo_tnode *tn2 = (coo_tnode*) rhs;
702  unsigned order = tn1->order;
703  for (unsigned i = 0; i < order && !comp; i++) {
704  comp = tn1->coords[i] - tn2->coords[i];
705  }

730  int read() {
731  // TNS files have no header, so this needs to be done in two passes.
732  // Pass 1:
733  // 1) Read first non-comment line and tokenize to get the order.
734  // 2) Read to end of file to get NNZ
735  // Pass 2:
736  // 1) Allocate the tensor.
737  // 2) Rewind the file pointer and read the indices and values.
738 
739  char line[1024];
740  char *pch;
741  bool found = 0;
742  bool mtx = 0;
743  unsigned lnum = 0;
744  unsigned pos = 0;
745  coo_tnode* nodes = NULL;
746 
747  FILE *in = fopen(_filename.c_str(), "r");
748  while (!found && fgets(line, sizeof(line), in) != NULL) {
749  found = (strlen(line) > 0 && !strchr(line, '#') && !strchr(line, '%'));
750  mtx |= (strstr(line, "MatrixMarket") != NULL);
751  }
752 
753  if (found) {
754  /* If MTX format, skip a line */
755  if (mtx) {
756  pch = fgets(line, sizeof(line), in);
757  }
758 
759  // Tokenize 1st line and get order
760  _order = 0;
761  for (pch = strtok(line, " "); pch != NULL; pch = strtok(NULL, " ")) {
762  _order += 1;
763  }
764  _order -= 1; // Subtract 1 for value
765 
766  // Count lines (nonzeros)
767  _nnz += 1;
768  while (fgets(line, sizeof(line), in) != NULL) {
769  if (strlen(line) > 0 && line[0] != '#') {
770  _nnz += 1;
771  }
772  }
773 
774  nodes = (coo_tnode*) calloc(_nnz, sizeof(coo_tnode));
775  for (unsigned i = 0; i < _nnz; i++) {
776  nodes[i].coords = (unsigned*) calloc(_order, sizeof(unsigned));
777  nodes[i].order = _order;
778  }
779  rewind(in);
780 
781  _indptr = (unsigned*) calloc(_nnz * _order, sizeof(unsigned));
782  _dims = (unsigned*) calloc(_order, sizeof(unsigned));
783  _vals = (dtype*) calloc(_nnz, sizeof(dtype));
784 
785  while (fgets(line, sizeof(line), in) != NULL) {
786  if (strlen(line) > 0 && line[0] != '#') {
787  pos = 0;
788  for (pch = strtok(line, " "); pch != NULL; pch = strtok(NULL, " ")) {
789  if (pos < _order) {
790  // offset2(i,j,N) (i)*(N)+(j)
791  unsigned ndx = (unsigned) atol(pch);
792  if (ndx > _dims[pos]) {
793  _dims[pos] = ndx;
794  }
795  nodes[lnum].coords[pos] = ndx - 1;
796  } else {
797  nodes[lnum].val = (dtype) atof(pch);
798  }
799  pos += 1;
800  }
801  lnum += 1;
802  }
803  }
804  }
805 
806  if (nodes != NULL) {
807  qsort(nodes, _nnz, sizeof(coo_tnode), coo_tnode_comp);
808 
809  for (unsigned i = 0; i < _nnz; i++) {
810  for (unsigned j = 0; j < _order; j++) {
811  _indptr[j * _nnz + i] = nodes[i].coords[j];
812  }
813  _vals[i] = nodes[i].val;
814  }
815 
816  for (unsigned i = 0; i < _nnz; i++) {
817  free(nodes[i].coords);
818  }
819  free(nodes);
820  }
821 
822  fclose(in);
823 
824  return _nnz < 1;
825  }
