16 void print_integer(int64_t value, buffered_ostream<CharT>& os)
17 {
18  CharT buf[255];
19  uint64_t u = (value < 0) ? static_cast<uint64_t>(-value) : static_cast<uint64_t>(value);
20  CharT* p = buf;
21  do
22  {
23  *p++ = static_cast<CharT>(48 + u%10);
24  }
25  while (u /= 10);
26  if (value < 0)

28  os.put('-');
29  }
30  while (--p >= buf)

32  os.put(*p);
